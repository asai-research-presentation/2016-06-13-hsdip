* Usage (on browsers)

#+BEGIN_SMALLER
+ n :: go to the next section/expand a list
+ p :: back to the previous section
+ s, or "go" :: jump to an arbitrary section (using minibuffer)
+ d :: debugging mode (shows content border)
+ <Ctrl-g> or <Esc> :: cancel current key strokes

#+END_SMALLER
